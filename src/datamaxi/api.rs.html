17/// The `User-Agent` sent with every request, e.g. `datamaxi-rust/0.4.0`.
18fn user_agent() -> String {
19    concat!("datamaxi-rust/", env!("CARGO_PKG_VERSION")).to_string()
20}

22/// Build the underlying blocking HTTP client with our defaults (timeout,
23/// `User-Agent`, unbounded idle pool). Falls back to a default client if the
24/// builder fails, so client construction is infallible and never panics.
25fn build_inner_client(timeout: Duration) -> reqwest::blocking::Client {
26    reqwest::blocking::Client::builder()
27        .pool_idle_timeout(None)
28        .timeout(timeout)
29        .user_agent(user_agent())
30        .build()
31        .unwrap_or_else(|_| reqwest::blocking::Client::new())
32}

60impl std::fmt::Debug for Client {
61    /// Redacts the API key so it never leaks into logs or error output.
62    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
63        f.debug_struct("Client")
64            .field("base_url", &self.base_url)
65            .field("api_key", &"<redacted>")
66            .finish_non_exhaustive()
67    }
68}

70impl Client {
71    /// Creates a new instance of the `Client` struct with the provided configuration.
72    ///
73    /// Uses the default timeout ([`DEFAULT_TIMEOUT`]). For more control over
74    /// timeout or reading the API key from the environment, use
75    /// [`ClientBuilder`].
76    pub fn new(config: Config) -> Self {
77        Client {
78            base_url: config.base_url.unwrap_or(BASE_URL.to_string()),
79            api_key: config.api_key,
80            inner_client: build_inner_client(DEFAULT_TIMEOUT),
81        }
82    }
83
84    /// Sends a GET request to the specified endpoint with optional parameters.
85    pub fn get<T: DeserializeOwned>(
86        &self,
87        endpoint: &'static str,
88        parameters: Option<HashMap<String, String>>,
89    ) -> Result<T> {
90        let url: String = format!("{}{}", self.base_url, endpoint);
91
92        let mut request = self
93            .inner_client
94            .get(url.as_str())
95            .header("X-DTMX-APIKEY", &self.api_key);
96
97        if let Some(p) = parameters {
98            request = request.query(&p);
99        }
100
101        let response = request.send()?;
102
103        self.handle_response(response)
104    }
105
106    /// Processes the response from the API and returns the result.
107    fn handle_response<T: DeserializeOwned>(&self, response: Response) -> Result<T> {
108        match response.status() {
109            StatusCode::OK => Ok(response.json::<T>()?),
110            StatusCode::INTERNAL_SERVER_ERROR => {
111                let mut response_text = String::new();
112                response.take(1000).read_to_string(&mut response_text)?;
113                Err(Error::InternalServerError(response_text))
114            }
115            StatusCode::UNAUTHORIZED => Err(Error::Unauthorized),
116            StatusCode::BAD_REQUEST => {
117                let mut response_text = String::new();
118                response.take(1000).read_to_string(&mut response_text)?;
119                Err(Error::BadRequest(response_text))
120            }
121            status => Err(Error::UnexpectedStatusCode(status.as_u16())),
122        }
123    }
124}

126/// Builder for a [`Client`], giving control over the API key source, base URL,
127/// and request timeout.
128///
129/// The API key may be provided explicitly via [`api_key`](ClientBuilder::api_key)
130/// or, if omitted, is read from the `DATAMAXI_API_KEY` environment variable at
131/// [`build`](ClientBuilder::build) time.
132///
133/// # Example
134/// ```no_run
135/// use datamaxi::api::ClientBuilder;
136/// use std::time::Duration;
137///
138/// // Explicit key + custom timeout.
139/// let client = ClientBuilder::new()
140///     .api_key("my_api_key")
141///     .timeout(Duration::from_secs(30))
142///     .build()
143///     .expect("api key provided");
144///
145/// // Key taken from the DATAMAXI_API_KEY environment variable.
146/// let client = ClientBuilder::new().build();
147/// ```
148#[derive(Debug, Clone)]
149pub struct ClientBuilder {
150    base_url: Option<String>,
151    api_key: Option<String>,
152    timeout: Duration,
153}
154
155impl ClientBuilder {
156    /// Creates a new builder with default settings (default timeout, key read
157    /// from the environment on `build`).
158    pub fn new() -> Self {
159        ClientBuilder {
160            base_url: None,
161            api_key: None,
162            timeout: DEFAULT_TIMEOUT,
163        }
164    }
165
166    /// Sets the API key explicitly, overriding the `DATAMAXI_API_KEY` environment variable.
167    pub fn api_key(mut self, api_key: impl Into<String>) -> Self {
168        self.api_key = Some(api_key.into());
169        self
170    }
171
172    /// Overrides the base URL (defaults to the production API).
173    pub fn base_url(mut self, base_url: impl Into<String>) -> Self {
174        self.base_url = Some(base_url.into());
175        self
176    }
177
178    /// Sets the per-request timeout (defaults to 10 seconds).
179    pub fn timeout(mut self, timeout: Duration) -> Self {
180        self.timeout = timeout;
181        self
182    }
183
184    /// Builds the [`Client`].
185    ///
186    /// Resolves the API key from the explicit value or the `DATAMAXI_API_KEY`
187    /// environment variable, returning [`Error::MissingApiKey`] if neither is set.
188    pub fn build(self) -> Result<Client> {
189        let api_key = self
190            .api_key
191            .or_else(|| std::env::var(API_KEY_ENV).ok())
192            .filter(|key| !key.trim().is_empty())
193            .ok_or(Error::MissingApiKey)?;
194
195        Ok(Client {
196            base_url: self.base_url.unwrap_or_else(|| BASE_URL.to_string()),
197            api_key,
198            inner_client: build_inner_client(self.timeout),
199        })
200    }
201}
202
203impl Default for ClientBuilder {
204    fn default() -> Self {
205        Self::new()
206    }
207}

212/// Errors returned by the Datamaxi+ API client.
213#[derive(Debug, Error)]
214pub enum Error {
215    /// No API key was provided explicitly and `DATAMAXI_API_KEY` is unset or empty.
216    #[error("missing API key: pass it to ClientBuilder::api_key or set DATAMAXI_API_KEY")]
217    MissingApiKey,
218
219    /// The API returned a `400 Bad Request`; the payload carries the server message.
220    #[error("Bad request: {0}")]
221    BadRequest(String),
222
223    /// The API returned a `401 Unauthorized` (missing or invalid API key).
224    #[error("Unauthorized")]
225    Unauthorized,
226
227    /// The API returned a `500 Internal Server Error`; the payload carries the server message.
228    #[error("Internal server error: {0}")]
229    InternalServerError(String),
230
231    /// The API returned a status code the client does not specifically handle.
232    #[error("Received unexpected status code: {0}")]
233    UnexpectedStatusCode(u16),
234
235    /// The underlying HTTP request failed, or the response body could not be decoded.
236    #[error(transparent)]
237    Http(#[from] reqwest::Error),
238
239    /// Reading the response body failed.
240    #[error(transparent)]
241    Io(#[from] std::io::Error),
242}
